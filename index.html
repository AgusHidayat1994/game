Lempar dadu.") } var isRolling by remember { mutableStateOf(false) } val coroutineScope = rememberCoroutineScope() // Fungsi Gerakan Pemain/Bot fun movePlayer(currentPos: Int, roll: Int, isBot: Boolean): Int { var newPos = currentPos + roll if (newPos > 100) { return currentPos // Harus pas ke 100 } // Cek Ular if (snakes.containsKey(newPos)) { newPos = snakes[newPos]!! if (!isBot) gameStatus = "Aww! Kamu digigit ular ke $newPos!" else gameStatus = "Yess! Bot digigit ular ke $newPos!" } // Cek Tangga else if (ladders.containsKey(newPos)) { newPos = ladders[newPos]!! if (!isBot) gameStatus = "Keren! Kamu naik tangga ke $newPos!" else gameStatus = "Oh tidak! Bot naik tangga ke $newPos!" } return newPos } // AI Bot Turn Logic LaunchedEffect(isPlayerTurn) { if (!isPlayerTurn && playerPos < 100 && botPos < 100) { gameStatus = "Bot sedang berpikir..." delay(1500) // Delay agar simulasi bot terasa nyata val botRoll = Random.nextInt(1, 7) diceRoll = botRoll botPos = movePlayer(botPos, botRoll, true) if (botPos == 100) { gameStatus = "Yahh, Bot Menang! Coba lagi." } else { gameStatus = "Bot mendapat angka $botRoll. Sekarang giliranmu!" isPlayerTurn = true } } } Column( modifier = Modifier .fillMaxSize() .padding(16.dp), horizontalAlignment = Alignment.CenterHorizontally, verticalArrangement = Arrangement.SpaceBetween ) { // Bagian Header Text( text = "ULAR TANGGA PRO", color = Color.Cyan, fontSize = 28.sp, fontWeight = FontWeight.Bold, modifier = Modifier.padding(vertical = 8.dp) ) // Status Game Text( text = gameStatus, color = Color.White, fontSize = 16.sp, textAlign = TextAlign.Center, modifier = Modifier .fillMaxWidth() .background(Color(0xFF1E1E1E), RoundedCornerShape(8.dp)) .padding(8.dp) ) // Papan Game (Board) Box( modifier = Modifier .aspectRatio(1f) .fillMaxWidth() .background(Color(0xFF1A1A1A), RoundedCornerShape(12.dp)) ) { GameBoardCanvas(playerPos = playerPos, botPos = botPos) } // Skor & Posisi Saat ini Row( modifier = Modifier.fillMaxWidth(), horizontalArrangement = Arrangement.SpaceEvenly ) { Text("Kamu (Merah): Posisi $playerPos", color = Color.Red, fontWeight = FontWeight.Bold) Text("Bot (Kuning): Posisi $botPos", color = Color.Yellow, fontWeight = FontWeight.Bold) } // Control Panel (Dadu & Tombol) Column( horizontalAlignment = Alignment.CenterHorizontally, modifier = Modifier.padding(bottom = 16.dp) ) { // Tampilan Dadu Keren Box( modifier = Modifier .size(60.dp) .background(Color.DarkGray, RoundedCornerShape(8.dp)), contentAlignment = Alignment.Center ) { Text( text = diceRoll.toString(), color = Color.White, fontSize = 32.sp, fontWeight = FontWeight.Bold ) } Spacer(modifier = Modifier.height(16.dp)) // Tombol Roll Dice Button( onClick = { if (isPlayerTurn && playerPos < 100 && botPos < 100 && !isRolling) { coroutineScope.launch { isRolling = true // Animasi acak dadu berputar cepat repeat(6) { diceRoll = Random.nextInt(1, 7) delay(100) } playerPos = movePlayer(playerPos, diceRoll, false) if (playerPos == 100) { gameStatus = "HOREEE! KAMU MENANG! 🎉" } else { isPlayerTurn = false } isRolling = false } } }, enabled = isPlayerTurn && playerPos < 100 && botPos < 100 && !isRolling, colors = ButtonDefaults.buttonColors(containerColor = Color(0xFF00E676)), shape = RoundedCornerShape(50), modifier = Modifier.width(200.dp) ) { Text( text = if (isPlayerTurn) "Kocok Dadu!" else "Bot Berjalan...", color = Color.Black, fontWeight = FontWeight.Bold, fontSize = 16.sp ) } } } } // Canvas untuk Menggambar Grid, Ular, Tangga, dan Pion @Composable fun GameBoardCanvas(playerPos: Int, botPos: Int) { Canvas(modifier = Modifier.fillMaxSize()) { val boardSize = 10 val cellSizeX = size.width / boardSize val cellSizeY = size.height / boardSize // 1. Gambar Grid (Papan 10x10) for (row in 0 until boardSize) { for (col in 0 until boardSize) { val isEvenRow = row % 2 == 0 val actualCol = if (isEvenRow) col else (boardSize - 1 - col) val cellNumber = (9 - row) * boardSize + (actualCol + 1) val x = col * cellSizeX val y = row * cellSizeY // Warna belang-belang keren (Retro cyberpunk) val cellColor = if ((row + col) % 2 == 0) Color(0xFF222222) else Color(0xFF2D2D2D) drawRect(color = cellColor, topLeft = Offset(x, y), size = androidx.compose.ui.geometry.Size(cellSizeX, cellSizeY)) } } // Fungsi bantu mendapatkan koordinat pixel dari nomor kotak (1-100) fun getCoordinates(position: Int): Offset { val zeroBasedPos = position - 1 val row = zeroBasedPos / 10 val col = zeroBasedPos % 10 val isEvenRow = row % 2 == 0 val targetCol = if (isEvenRow) col else (10 - 1 - col) val targetRow = 9 - row return Offset( x = targetCol * cellSizeX + (cellSizeX / 2), y = targetRow * cellSizeY + (cellSizeY / 2) ) } // 2. Gambar Tangga (Warna Hijau Neon) ladders.forEach { (start, end) -> val startPt = getCoordinates(start) val endPt = getCoordinates(end) drawLine( color = Color(0xFF00FFCC), start = startPt, end = endPt, strokeWidth = 8f, cap = StrokeCap.Round ) } // 3. Gambar Ular (Warna Merah Neon/Orange) snakes.forEach { (start, end) -> val startPt = getCoordinates(start) // Kepala Ular val endPt = getCoordinates(end) // Ekor Ular drawLine( color = Color(0xFFFF3366), start = startPt, end = endPt, strokeWidth = 10f, cap = StrokeCap.Round ) } // 4. Gambar Pion Player (Lingkaran Merah) val playerCoords = getCoordinates(playerPos) drawCircle( color = Color.Red, radius = cellSizeX / 4f, center = Offset(playerCoords.x - 10f, playerCoords.y - 10f)
